query CharactersQuery(
  $showUser: Int!
  $searchUser: String
) {
  character (id: $showUser) {
    name
    age
  }
  characters(search: $searchUser) {
    id
    name
  }
}

mutation CreateCharacterMutation (
  $name: String!
  $age: Int!
) {
  createCharacter(
    name: $name
    age: $age
  ) {
    id
    name
    age
  }
}
